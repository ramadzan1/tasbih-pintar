  <script>

    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

    function playBeep(freq = 600, duration = 0.1) {

      if (!document.getElementById('beepFeedbackToggle').checked) return;

      try {

        const osc = audioCtx.createOscillator();

        const gain = audioCtx.createGain();

        osc.connect(gain);

        gain.connect(audioCtx.destination);

        osc.frequency.value = freq;

        gain.gain.setValueAtTime(0.1, audioCtx.currentTime);

        gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);

        osc.start();

        osc.stop(audioCtx.currentTime + duration);

      } catch (e) {

        console.log("AudioContext error: ", e);

      }

    }


    function playSuccessSound() {

      playBeep(880, 0.15);

      setTimeout(() => playBeep(1100, 0.25), 100);

    }

    // Kira % Keseluruhan Misi Harian

    function calculateOverallProgress() {

      const totalCount = zikirData.reduce((acc, curr) => acc + Math.min(curr.count, curr.target), 0);

      const totalTarget = zikirData.reduce((acc, curr) => acc + curr.target, 0);

      const overallPercent = totalTarget > 0 ? Math.round((totalCount / totalTarget) * 100) : 0;


      document.getElementById('overallProgressPercent').innerText = `${overallPercent}% Selesai`;

      document.getElementById('overallProgressBar').style.width = `${overallPercent}%`;

    }


    // Kemaskini sasaran dinamik oleh user

    function updateTarget(id, value) {

      const targetVal = parseInt(value);

      if (isNaN(targetVal) || targetVal < 1) return;

      

      const zikir = zikirData.find(z => z.id === id);

      if (zikir) {

        zikir.target = targetVal;

        if (zikir.count >= zikir.target && !zikir.completed) {

          zikir.completed = true;

        } else if (zikir.count < zikir.target) {

          zikir.completed = false;

        }

        saveToLocalStorage();

        renderZikirCards();

      }

    }


    // Tolak atau Tambah Bilangan Secara Manual (Sangat mudah kalau voice detect tersilap)

    function adjustCount(id, amount) {

      const zikir = zikirData.find(z => z.id === id);

      if (zikir) {

        zikir.count = Math.max(0, zikir.count + amount);

        

        if (zikir.count >= zikir.target && !zikir.completed) {

          zikir.completed = true;

          playSuccessSound();

          showToast(`Sasaran ${zikir.name} tercapai!`);

        } else if (zikir.count < zikir.target) {

          zikir.completed = false;

        }

        

        saveToLocalStorage();

        renderZikirCards();

        playBeep(amount > 0 ? 500 : 350, 0.08);

      }

    }

      toast.classList.remove('opacity-0', 'pointer-events-none', 'translate-y-4');

      toast.classList.add('opacity-100', 'translate-y-0');


      setTimeout(() => {

        toast.classList.add('opacity-0', 'pointer-events-none', 'translate-y-4');

        toast.classList.remove('opacity-100', 'translate-y-0');

      }, 3000);

    // Setup Pengecaman Suara Web Speech API

    function setupSpeechRecognition() {

      const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

      

      if (!SpeechRecognition) {

        document.getElementById('engineStatus').innerText = "Ralat: Browser tidak menyokong fungsi suara.";

        document.getElementById('masterActionBtn').disabled = true;

        return;

      }


      recognition = new SpeechRecognition();

      recognition.continuous = true;

      recognition.lang = 'ms-MY'; // Set Bahasa Melayu untuk fonetik tempatan yang lebih tepat

      recognition.interimResults = false;


      recognition.onstart = () => {

        isListening = true;

        document.getElementById('engineStatus').innerText = "Sedang mendengar zikir anda...";

        updateButtonUI(true);

      };


      recognition.onresult = (event) => {

        const lastResultIndex = event.results.length - 1;

        const textSpoken = event.results[lastResultIndex][0].transcript.toLowerCase().trim();

        

        document.getElementById('engineStatus').innerText = `Didengar: "${textSpoken}"`;

        processSpokenWords(textSpoken);

      };


      recognition.onerror = (event) => {

        console.error("Speech Recognition Error", event.error);

        if (event.error === 'not-allowed') {

          document.getElementById('engineStatus').innerText = "Akses mikrofon ditolak.";

          stopListening();

        }

      };


      recognition.onend = () => {

        // Auto-restart demi kemudahan memandu jika user tidak menekan butang henti sendiri

        if (isListening) {

          try {

            recognition.start();

          } catch (e) {

            console.log("Error restarting recognition:", e);

          }

        }

      };

    }


    // Proses kata yang disebut dan tentukan kaunter zikir mana yang perlu dinaikkan

    function processSpokenWords(text) {

      let detected = false;


      for (let zikir of zikirData) {

        // Semak sekiranya teks mengandungi mana-mana kata kunci yang diprogramkan

        const matched = zikir.keywords.some(keyword => text.includes(keyword));

        

        if (matched) {

          detected = true;

          zikir.count++;

          

          playBeep(650, 0.08); // Bunyi feedback ringkas bahawa perkataan berjaya dikesan

          

          if (zikir.count >= zikir.target && !zikir.completed) {

            zikir.completed = true;

            playSuccessSound();

            speakText(`${zikir.name} selesai`);

            showToast(`Misi ${zikir.name} tercapai!`);

          } else {

            // Sebut nombor kaunter terkini

            speakText(zikir.count.toString());

          }

          

          saveToLocalStorage();

          renderZikirCards();

          break; // Berhenti sekiranya sudah jumpa padanan pertama

        }

      }


      if (!detected) {

        // Jika tidak dapat kesan kata kunci dengan tepat

        console.log("No matching zikir for: ", text);

      }

    }


    // Kawalan Butang Hidup/Matikan Pengecaman

    function toggleListening() {

      if (!recognition) {

        setupSpeechRecognition();

      }


      if (isListening) {

        stopListening();

      } else {

        startListening();

      }

    }


    function startListening() {

      // Minta kebenaran AudioContext dahulu sekiranya disekat

      if (audioCtx.state === 'suspended') {

        audioCtx.resume();

      }

      try {

        recognition.start();

      } catch (e) {

        console.log("Start error: ", e);

      }

    }


    function stopListening() {

      isListening = false;

      if (recognition) {

        recognition.stop();

      }

      document.getElementById('engineStatus').innerText = "Sesi dihentikan.";

      updateButtonUI(false);

    }

    // Kekalkan Skrin Hidup (Wake Lock API) bagi mengelakkan auto-lock semasa memandu

    let wakeLock = null;

    async function requestWakeLock() {

      try {

        if ('wakeLock' in navigator) {

          wakeLock = await navigator.wakeLock.request('screen');

          console.log('Screen Wake Lock is active!');

        }

      } catch (err) {

        console.warn(`${err.name}, ${err.message}`);

      }

    }


    // Aktifkan wake lock setiap kali window mendapat fokus semula

    document.addEventListener('visibilitychange', async () => {

      if (wakeLock !== null && document.visibilityState === 'visible') {

        await requestWakeLock();

      }

    });


    // Mula Aplikasi semasa tetingkap dimuatkan

    window.onload = () => {

      renderZikirCards();

      requestWakeLock();


      // Sekiranya persekitaran tidak membenarkan Speech Synthesis, tutup pilihan visual tersebut dengan selamat

      if (!hasSpeechSynthesis) {

        const voiceToggle = document.getElementById('voiceFeedbackToggle');

          voiceToggle.checked = false;

          voiceToggle.disabled = true;

          const parentLabel = voiceToggle.closest('label');

          if (parentLabel) {

            parentLabel.classList.add('opacity-40', 'cursor-not-allowed');

            const spanText = parentLabel.querySelector('span');

            if (spanText) spanText.innerText = "Sebut Bilangan (Tiada Sokongan)";

          }

        }

      }

    };
